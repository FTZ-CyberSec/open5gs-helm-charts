$(kubectl get svc --namespace {{ .Release.Namespace }} {{ template "open5gs.fullname" . }}-mme -o jsonpath='{.spec.clusterIP}')

{{- if .Values.webui.ingress.enabled }}

  {{- if $host.tls }}
  https://{{ $host.name }}
  {{- else }}
  http://{{ $host.name }}
  {{- end }}
{{- end }}
{{- end }}
